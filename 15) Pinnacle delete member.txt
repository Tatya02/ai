#include <iostream>
#include <string>
using namespace std;

// Node structure for club members
struct Node {
    string name;
    long long prn;
    Node *next;
};

// Class for Pinnacle Club
class PinnacleClub {
    Node *head;

public:
    PinnacleClub() { head = NULL; }

    // Function to add President (at beginning)
    void addPresident(long long prn, string name) {
        Node *temp = new Node;
        temp->prn = prn;
        temp->name = name;
        temp->next = head;
        head = temp;
        cout << "âœ… President added successfully.\n";
    }

    // Function to add Secretary (at end)
    void addSecretary(long long prn, string name) {
        Node *temp = new Node;
        temp->prn = prn;
        temp->name = name;
        temp->next = NULL;

        if (head == NULL) {
            head = temp;
        } else {
            Node *p = head;
            while (p->next != NULL)
                p = p->next;
            p->next = temp;
        }
        cout << "âœ… Secretary added successfully.\n";
    }

    // Function to add Member (between President and Secretary)
    void addMember(long long prn, string name) {
        if (head == NULL) {
            cout << "âš  Please add the President first!\n";
            return;
        }

        Node *temp = new Node;
        temp->prn = prn;
        temp->name = name;
        temp->next = NULL;

        Node *p = head;

        // Insert before Secretary
        while (p->next != NULL && p->next->next != NULL)
            p = p->next;

        temp->next = p->next;
        p->next = temp;

        cout << "âœ… Member added successfully.\n";
    }

    // Function to delete a member by PRN
    void deleteMember(long long prn) {
        if (head == NULL) {
            cout << "âš  Club is empty.\n";
            return;
        }

        Node *temp = head;

        // Deleting President
        if (head->prn == prn) {
            head = head->next;
            delete temp;
            cout << "ðŸ—‘ President deleted successfully.\n";
            return;
        }

        Node *prev = NULL;
        while (temp != NULL && temp->prn != prn) {
            prev = temp;
            temp = temp->next;
        }

        if (temp == NULL) {
            cout << "âŒ Member not found.\n";
            return;
        }

        prev->next = temp->next;
        delete temp;

        if (prev->next == NULL)
            cout << "ðŸ—‘ Secretary deleted successfully.\n";
        else
            cout << "ðŸ—‘ Member deleted successfully.\n";
    }

    // Function to display all members
    void display() {
        if (head == NULL) {
            cout << "âš  Club has no members.\n";
            return;
        }

        Node *p = head;
        cout << "\n--- Pinnacle Club Members ---\n";
        while (p != NULL) {
            if (p == head)
                cout << "ðŸ‘‘ [President] ";
            else if (p->next == NULL)
                cout << "ðŸ“œ [Secretary] ";
            else
                cout << "ðŸ‘¥ [Member] ";
            cout << "PRN: " << p->prn << " | Name: " << p->name << endl;
            p = p->next;
        }
    }

    // Function to concatenate another divisionâ€™s list
    void concatenate(PinnacleClub &other) {
        if (head == NULL) {
            head = other.head;
            return;
        }

        Node *p = head;
        while (p->next != NULL)
            p = p->next;
        p->next = other.head;

        cout << "\nâœ… Two divisions concatenated successfully.\n";
    }
};
