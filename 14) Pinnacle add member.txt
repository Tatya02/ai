#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Structure to store member info
struct Node {
    int prn;
    char name[50];
    struct Node* next;
};

// Head and tail pointers
struct Node* head = NULL;  // President
struct Node* tail = NULL;  // Secretary

// Function to create a new node
struct Node* createNode(int prn, char name[]) {
    struct Node* newNode = (struct Node*)malloc(sizeof(struct Node));
    newNode->prn = prn;
    strcpy(newNode->name, name);
    newNode->next = NULL;
    return newNode;
}

// Add president (at beginning)
void addPresident(int prn, char name[]) {
    if (head != NULL) {
        printf("âš  President already exists!\n");
        return;
    }
    struct Node* newNode = createNode(prn, name);
    head = tail = newNode;
    printf("âœ… President added: %s (PRN: %d)\n", name, prn);
}

// Add secretary (at end)
void addSecretary(int prn, char name[]) {
    if (head == NULL) {
        printf("âš  Add president first!\n");
        return;
    }
    if (tail != NULL && tail != head && tail->next == NULL) {
        // already has secretary
        printf("âš  Secretary already exists!\n");
        return;
    }
    struct Node* newNode = createNode(prn, name);
    tail->next = newNode;
    tail = newNode;
    printf("âœ… Secretary added: %s (PRN: %d)\n", name, prn);
}

// Add member (between president and secretary)
void addMember(int prn, char name[]) {
    if (head == NULL) {
        printf("âš  Add president first!\n");
        return;
    }
    struct Node* newNode = createNode(prn, name);

    // if only president exists
    if (head == tail) {
        head->next = newNode;
        tail = newNode;
    } else {
        struct Node* temp = head;
        while (temp->next != tail && temp->next != NULL)
            temp = temp->next;
        newNode->next = temp->next;
        temp->next = newNode;
    }

    printf("âœ… Member added: %s (PRN: %d)\n", name, prn);
}

        else

// Count total members
void totalMembers() {
    int count = 0;
    struct Node* temp = head;
    while (temp != NULL) {
        count++;
        temp = temp->next;
    }
    printf("\nğŸ‘¥ Total members (including President & Secretary): %d\n", count);
}
